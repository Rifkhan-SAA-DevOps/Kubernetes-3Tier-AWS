4. Backend in Kubernetes (Real 3-Tier Logic)
    backend-secret.yaml
    kubectl apply -f k8s/backend/backend-secret.yaml

    Create Docker File For Backend
    minikube docker-env | Invoke-Expression  => point 
    docker build -t backend:v1 .

    backend-deployment.yaml
    kubectl apply -f k8s/backend/backend-deployment.yaml

    backend-service.yaml
    kubectl apply -f k8s/backend/backend-service.yaml

    kubectl apply -f k8s/backend/

    kubectl delete deployment backend -n three-tier-app

    kubectl describe pod backend-79b4d87dd5-mksv8 -n three-tier-app

    kubectl rollout restart deployment backend -n three-tier-app
    kubectl rollout restart deployment frontend -n three-tier-app



5. Frontend (React) in Kubernetes
    Create Docker File For Frontend
    minikube docker-env | Invoke-Expression  => point 
    docker build -t frontend:v1 .

    frontend-deployment.yaml
    kubectl apply -f k8s/frontend/frontend-deployment.yaml

    frontend-service.yaml
    kubectl apply -f k8s/frontend/frontend-service.yaml


    kubectl get svc -n three-tier-app

    minikube service frontend-service -n three-tier-app  =>  to open app

    kubectl scale deployment backend --replicas=0 -n three-tier-app
    kubectl scale deployment frontend --replicas=0 -n three-tier-app
    kubectl scale deployment mysql --replicas=0 -n three-tier-app

    kubectl delete deployment backend -n three-tier-app
    kubectl delete deployment frontend -n three-tier-app
    kubectl delete deployment mysql -n three-tier-app

    minikube stop
    minikube start
